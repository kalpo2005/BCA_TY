1. Create a table 
	CREATE TABLE customer(
		cust_id NUMBER(4),
		cust_city VARCHAR2(15),
		product_name VARCHAR2(5),
		product_price NUMBER(7,2) NOT NULL,
		quantity NUMBER(5) NOT NULL,
		bill_amt NUMBER(7,2) NOT NULL,
		PRIMARY KEY(cust_id)
	);
