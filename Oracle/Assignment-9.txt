SECOND TABLE 

	CREATE TABLE student_membar(
		roll_no NUMBER(3),
		fname VARCHAR2(20),
		mname VARCHAR2(20),
		sname VARCHAR2(20),
		dept_id NUMBER(10),
		semester NUMBER(3),
		contact_no NUMBER(10),
		gender CHAR(1),
		PRIMARY KEY(roll_no),
		FOREIGN KEY (dept_id) REFERENCES department(dept_id),
		CHECK (gender IN ('M','F'))
	);
